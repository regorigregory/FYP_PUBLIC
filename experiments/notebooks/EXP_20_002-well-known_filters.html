filters = [laplace, median, meijering, sato, frangi, hessian, prewitt, sobel, scharr]

fig = plt.figure()
plt.subplots(3,3, figsize=[20, 10])
plt.subplots_adjust(hspace=0.5, wspace=0.5)
plot_counter = 251
for i, filter in enumerate(filters):
    cvt_img = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    result = filter(cvt_img)
    ax = plt.subplot(plot_counter)
    plot_counter+=1
    ax.set_title(filter.__name__)
    plt.imshow(result, cmap="gray")
